package main

import (
	"fmt"
	"io"
	"strings"
)

func main() {
	input := "Пример текста\nС другой строкой\nИ так далее..."
	reader := strings.NewReader(input)

	// Создаем буфер для чтения данных
	buffer := make([]byte, 20)

	for {
		// Читаем данные из потока в буфер
		n, err := reader.Read(buffer)

		if err == io.EOF {
			// Достигнут конец файла (End of File)
			break
		}

		if err != nil {
			fmt.Printf("Ошибка чтения: %v\n", err)
			break
		}

		// Обработка данных (например, вывод на экран)
		fmt.Printf("Прочитано %d байт: %s\n", n, buffer[:n])
	}
}
